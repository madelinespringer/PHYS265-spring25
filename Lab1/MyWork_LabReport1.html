# Import required Python environments
import matplotlib.pyplot as plt
import numpy as np 
import math

# Create table of constants
G = 6.67*(10**-11) # Gravitational Constant
g = 9.81 # Gravitational Acceleration
Me = 5.9*(10**24) # Mass of Earth
Mm = 7.3*(10**22) # Mass of Moon
Ma = 5500 # Mass of Apollo Command Module
Re = 6378*(10**3) # Radius of Earth
Rm = 1737*(10**3) # Radius of Moon
D = 3.8*(10**8) # Distance from Earth to Moon
Ve = 2.4*(10**3) # Exhaust Velocity of Saturn V Stage 1 (S1-C)
m = 1.3*(10**4) # Burn Rate of S1-C
m0 = 2.8*(10**6) # Wet mass of S-1C
mf = 7.5*(10**5) # Dry mass of S-1C

# Gravitational Potential Function
def grav_potential(Me, xm, ym, x, y):
    r = np.sqrt((x-xm)**2 - (y-ym)**2)
    r[r == 0] = np.nan
    return (-G*Me)/r

# Plot of Gravitational Potential v. Distance From Earth 
fig, ax = plt.subplots()
x1 = np.linspace(0, 1.5*D, 100)
ax.plot(x1, np.abs(grav_potential(Me, 0, 0, x1, 0)))
ax.set_yscale('log')

ax.set_xlabel('Distance x From Earth (m)')
ax.set_ylabel('Absolute Value of Gravitational Potential (m**2/s**2)')
ax.set_title('Gravitational Potential v. Distance From Earth')

def egp(Me, xm, ym, x, y):
    r = np.sqrt((x-xm)**2 + (y-ym)**2)
    r[r == 0] = np.nan
    return (-G*Me)/r

def mgp(Mm, xm, ym, x, y):
    r = np.sqrt((x-xm)**2 + (y-ym)**2)
    r[r == 0] = np.nan
    return (-G*Mm)/r

## Gravitational Force Function
from matplotlib.collections import LineCollection
def grav_force(M1, m2, x1, y1, x2, y2): 
    r = np.sqrt((x2-x1)**2+(y2-y1)**2) 
    r = np.where(r == Re, np.nan, r)
    total_force = G*M1*m2/r**2 
    total_force = np.where(r<Re, np.nan, total_force)
    Fx = total_force*(x2-x1)/r 
    Fy = total_force*(y2-y1)/r 
    return Fx, Fy 

## Streamplot of Gravitational Force 
fig, ax3 = plt.subplots() 
x3 = np.linspace(-1.5*D, 1.5*D, 100) 
y3 = np.copy(x3) 
x_grid, y_grid = np.meshgrid(x3,y3) 

Fx = grav_force(Me, Ma, 0, 0, x_grid, y_grid)[0]+grav_force(Mm, Ma, D/np.sqrt(2), D/np.sqrt(2), x_grid, y_grid)[0] 
Fy = grav_force(Me, Ma, 0, 0, x_grid, y_grid)[1]+grav_force(Mm, Ma, D/np.sqrt(2), D/np.sqrt(2), x_grid, y_grid)[1] 

colors = np.sqrt(Fx**2+Fy**2) 
stream = ax3.streamplot(x_grid, y_grid, Fx, Fy, color = colors, cmap = cm.RdPu, norm = LogNorm()) 
cbar = fig.colorbar(stream.lines)  
ax3.set_aspect('equal') 

ax3.set_xlabel('x distance from Earth (m)') 
ax3.set_ylabel('y distance from Earth (m)') 
ax3.set_title('2D streamplot of Gravitational Potential in the Earth and Moon System')
cbar.set_label('Gravitational Force (N)')

## Calculate Burn Time
T = (m0-mf)/m
print('S-IC Burn Time (T): 'f'{T:0.3g}', 'seconds')

## Velocity Change Formula
def vel_change(t): 
    mt = m0 - (m0 - mf) * (t/T) 
    if mt > mf: 
        return Ve*np.log(m0/mt)-g*t
    else: 
        return Ve*np.log(m0/mf)-g*t

vel_final = vel_change(T) 
print('S-IC Velocity Change after T: 'f'{vel_final:0.3g}','meters per second')

## Rocket's Altitude After Burn 
def vel_change(t):
    mt = m0-m*t 
    mt = np.where(mt<mf, mf, mt)  
    vel_change_more_t = Ve*np.log(m0/mf)-g*T-g*(t-T)  
    vel_change_t = np.where(t>T, vel_change_more_t, Ve*np.log(m0-(mt))-g*t)
    return vel_change_t 
print('S-IC Altitude after T, Method 1: 'f'{height:0.3g}','meters')
